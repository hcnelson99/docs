\section{Semantics}

\begin{figure}[h]
\hfill \fbox{$A; \Sigma; \Gamma \vdash \tau <: \tau'$}
\begin{mathpar}
\inferrule
	{}
	{A, \Sigma; \Gamma \vdash \tau \; \texttt{<:}\; \tau}
	\quad (\textsc {S-Refl})
	\and
\inferrule
	{}
	{\Sigma; \Gamma \vdash \tau\{z \Rightarrow \overline{\sigma}\}\; <:\; \tau}
	\quad (\textsc {S-Refine})
	\and
\inferrule
	{\Sigma; \Gamma \vdash x \ni \texttt{type} \; M \; L : \tau_1 .. \tau_2\\
	 \Sigma; \Gamma \vdash \tau_2 <: \tau}
	{\Sigma; \Gamma \vdash x.L\; <:\; \tau}
	\quad (\textsc {S-Select-Upper})
	\and
\inferrule
	{\Sigma; \Gamma \vdash x \ni \texttt{type} \; M \; L : \tau_1 .. \tau_2\\
	 \Sigma; \Gamma \vdash \tau <: \tau_1}
	{\Sigma; \Gamma \vdash \tau \; <:\; x.L}
	\quad (\textsc {S-Select-Lower})
	\and
\inferrule
	{\Sigma; \Gamma \vdash \tau \; \texttt{<:}\; \tau_1}
	{\Sigma; \Gamma \vdash \tau \; \texttt{<:}\; \tau_1 \vee \tau_2}
	\quad (\textsc {S-Union-L})
	\and
\inferrule
	{\Sigma; \Gamma \vdash \tau \; \texttt{<:}\; \tau_2}
	{\Sigma; \Gamma \vdash \tau \; \texttt{<:}\; \tau_1 \vee \tau_2}
	\quad (\textsc {S-Union-R})
	\and
\inferrule
	{\Sigma; \Gamma \vdash \tau \; \texttt{<:}\; \tau_1 \\
	 \Sigma; \Gamma \vdash \tau \; \texttt{<:}\; \tau_2}
	{\Sigma; \Gamma \vdash \tau \; \texttt{<:}\; \tau_1 \wedge \tau_2}
	\quad (\textsc {S-Intersection})
	\and
\inferrule
	{}
	{\Sigma; \Gamma \vdash \tau \; \texttt{<:}\; \top}
	\quad (\textsc {S-Top})
	\and
\inferrule
	{}
	{\Sigma; \Gamma \vdash \bot \; \texttt{<:}\; \tau}
	\quad (\textsc {S-Bottom})
\end{mathpar}
\hfill \fbox{$A; \Sigma; \Gamma \vdash \sigma <: \sigma'$}
\begin{mathpar}
\inferrule
	{}
	{\Sigma; \Gamma \vdash \texttt{val} \; f:T <: \texttt{val} \; f:T}
	\quad (\textsc {S-Decl-Val})
	\and
\inferrule
	{\Sigma; \Gamma \vdash S' <: S \\
	 \Sigma; \Gamma \vdash T <: T'}
	{\Sigma; \Gamma \vdash \texttt{def} \; m:S \rightarrow T <: \texttt{def} \; m:S' \rightarrow T'}
	\quad (\textsc {S-Decl-Def})
	\and
\inferrule
	{\Sigma; \Gamma \vdash S' <: S \\
	 \Sigma; \Gamma \vdash U <: U'}
	{\Sigma; \Gamma \vdash \texttt{type} \; L : S .. U \; <:\; \texttt{type} \; L : S' .. U'}
	\quad (\textsc {S-Decl-Type})
\end{mathpar}
\caption{Subtyping}
\label{f:subtype}
\end{figure}

\begin{figure}[h]
\hfill \fbox{$A; \Sigma; \Gamma \vdash T \;  \textbf{wf}$}
\begin{mathpar}
\inferrule
  {\Sigma; \Gamma \vdash p \ni \texttt{type} \; L : S .. U \\
  	\Sigma; \Gamma \vdash \texttt{type} \; L : S .. U \; \textbf{wf}}
  {\Sigma; \Gamma \vdash p.L \; \textbf{wf}}
  \quad (\textsc {WF-Sel})
	\and
\inferrule
  {\Gamma; \Sigma \vdash \tau \prec \overline{\sigma}' \\
   \forall \sigma \in \overline{\sigma}, \exists \sigma' \in \overline{\sigma}' s.t. \; name(\sigma') = name(\sigma) \Longrightarrow \Gamma \vdash \sigma <: \sigma'}
  {\Gamma; \Sigma \vdash \tau\{z \Rightarrow \overline{\sigma}\} \; \textbf{wf}}
  \quad (\textsc {WF-Refine})
	\and
\inferrule
  {\Sigma; \Gamma \vdash \tau_1 \;  \textbf{wf} \\
   \Sigma; \Gamma \vdash \tau_2 \;  \textbf{wf}}
  {\Sigma; \Gamma \vdash \tau_1 \wedge \tau_2 \;  \textbf{wf}}
  \quad (\textsc {WF-Intersection})
	\and
\inferrule
  {\Sigma; \Gamma \vdash \tau_1 \;  \textbf{wf} \\
   \Sigma; \Gamma \vdash \tau_2 \;  \textbf{wf}}
  {\Sigma; \Gamma \vdash \tau_1 \vee \tau_2 \;  \textbf{wf}}
  \quad (\textsc {WF-Union})
	\and
\inferrule
  {}
  {\Sigma; \Gamma \vdash \top \;  \textbf{wf}}
  \quad (\textsc {WF-Top})
	\and
\inferrule
  {}
  {\Sigma; \Gamma \vdash \bot \;  \textbf{wf}}
  \quad (\textsc {WF-Bottom})
\end{mathpar}
\hfill \fbox{$A; \Sigma; \Gamma \vdash \sigma \;  \textbf{wf}$}
\begin{mathpar}
\inferrule
  {A; \Sigma; \Gamma \vdash T : \textbf{wf}}
  {A; \Sigma; \Gamma \vdash \texttt{val} \; f:T \;  \textbf{wf}}
  \quad (\textsc {WF-Val})
	\and
\inferrule
  {A; \Sigma; \Gamma \vdash T : \textbf{wf} \\
  	A; \Sigma; \Gamma \vdash S : \textbf{wf}}
  {A; \Sigma; \Gamma \vdash \texttt{def} \; m:S \rightarrow T \;  \textbf{wf}}
  \quad (\textsc {WF-Def})
	\and
\inferrule
  {A; \Sigma; \Gamma \vdash S : \textbf{wfe} \; \vee \; S = \bot\\
  	A; \Sigma; \Gamma \vdash U : \textbf{wfe} \\
  	A; \Sigma; \Gamma \vdash S <: U}
  {A; \Sigma; \Gamma \vdash \texttt{type} \; L : S .. U \; \textbf{wf}}
  \quad (\textsc {WF-Type})
\end{mathpar}
\hfill \fbox{$A; \Sigma \vdash \Gamma \;  \textbf{wf}$}
\begin{mathpar}
\inferrule
  {\forall x \in dom(\Gamma), A; \Sigma; \Gamma \vdash \Gamma(x) \; \textbf{wf}}
  {\Sigma \vdash \Gamma \;  \textbf{wf}}
  \quad (\textsc {WF-Environment})
\end{mathpar}
\hfill \fbox{$\Sigma \;  \textbf{wf}$}
\begin{mathpar}
\inferrule
  {\forall l \in dom(\Sigma), \varnothing; \Sigma; \varnothing \vdash \Sigma(l) \; \textbf{wf}}
  {\Sigma \;  \textbf{wf}}
  \quad (\textsc {WF-Store-Context})
\end{mathpar}
\begin{mathpar}
\inferrule
  {\forall l \in dom(\mu), \varnothing; \Sigma; \varnothing \vdash \mu(l) : \Sigma(l)}
  {\mu : \Sigma}
  \quad (\textsc {WF-Store})
\end{mathpar}
\caption{Well-Formedness}
\label{f:wf}
\end{figure}

\begin{mathpar}
\inferrule
  {x \in dom(\Gamma)}
  {	A; \Sigma; \Gamma \vdash x : \Gamma(x)}
  \quad (\textsc {T-Var})
	\and
\inferrule
  {	l \in dom(\Sigma)}
  {	A; \Sigma; \Gamma \vdash l : \Sigma(l)}
  \quad (\textsc {T-Loc})
	\and
\inferrule
  {A; \Sigma; \Gamma, z : \{z \Rightarrow \overline{\sigma}\} 
  \vdash \overline{d} : \overline{\sigma}}
  {A; \Sigma; \Gamma \vdash \texttt{new} \; \{z \Rightarrow \overline{d}\} : 
  \{z \Rightarrow \overline{\sigma}\}}
  \quad (\textsc {T-New})
	\and
\inferrule
  {A; \Sigma; \Gamma \vdash e_0 \ni \texttt{def} \; m:S \rightarrow T \\
  	A; \Sigma; \Gamma \vdash e_0 : T_0 \\
  	A; \Sigma; \Gamma \vdash e_1 : S' \\
  	A; \Sigma; \Gamma \vdash S' <: S}
  {A; 	\Sigma; \Gamma \vdash e_0.m(e_1) : T}
  \quad (\textsc {T-Meth})
	\and
\inferrule
  {	A; \Sigma; \Gamma \vdash e : S \\
  	A; \Sigma; \Gamma \vdash e \ni \texttt{val} \; f:T}
  {	A; \Sigma; \Gamma \vdash e.f : T}
  \quad (\textsc {T-Acc})
	\and
\inferrule
  {A; \Sigma; \Gamma \vdash e : S \\
   A; \Sigma; \Gamma \vdash S <: T}
  {A; \Sigma; \Gamma \vdash e \unlhd T : T}
  \quad (\textsc {T-Type})
